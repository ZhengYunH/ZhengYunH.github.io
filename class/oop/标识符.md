## 友元 friend
```
class A
{
    //可以直接访问该类的private and protected成员
    friend void func();
    friend class B;
    friend voif C::f();

private:
    int x;
}

B.x=1;
```
## 常元 const  
利用const修饰函数和变量
```
class A
{
    int x;
    char *p;
public:
    int get_day() const;
    int set(int year,int month,int day);
    void f() const{
        x=10;    //error
        p=new char[20];    //error
        strcpy(p,"ABCD");   //feasible,doesn't change the value of p
    }
}

void f(const Date &d){
    d.get_day();    //OK
    d.set(11,1,1);  //error
}

```
## 静态 static  
静态成员只有一个拷贝（即只有一个空间）
```
class A
{
    static int x;
    void f(){x++;}
    static int get_X();//只能访问静态成员
}

int A::x=0;
A a,b;
a.f();
b.f();
cout<<A::get_shared();//feasible
```
